**Reducer (exported for testing):**
```typescript
export function centeringReducer(state: CenteringState, action: CenteringAction): CenteringState {
  switch (action.type) {
    case 'GPS_LOCK':
      return { ...state, mode: 'centered', userPosition: action.position };
    case 'USER_PAN':
      if (state.mode === 'initializing') return state;
      return { ...state, mode: 'free-pan' };
    case 'RECENTER':
      return { ...state, mode: 'centered' };
    case 'START_NAVIGATION':
      return { ...state, mode: 'navigating' };
    case 'STOP_NAVIGATION':
      return { ...state, mode: 'centered' };
    case 'GPS_UPDATE':
      return { ...state, userPosition: action.position };
    default:
      return state;
  }
}
```

**Hook:**
```typescript
export function useMapCentering() {
  const [state, dispatch] = useReducer(centeringReducer, {
    mode: 'initializing',
    userPosition: null,
    lastStoredPosition: null,
  });

  // Load last-known position from IndexedDB on mount
  useEffect(() => {
    dbGet<{ key: string; lng: number; lat: number; timestamp: number }>('settings', 'lastPosition')
      .then((stored) => {
        if (stored && (Date.now() - stored.timestamp < 24 * 60 * 60 * 1000)) {
          dispatch({ type: 'GPS_UPDATE', position: [stored.lng, stored.lat] }); // Don't use GPS_LOCK — this is stored, not live
          // Also track it as lastStoredPosition for map init
          // (handled via separate state update or ref)
        }
      })
      .catch(() => { /* no stored position, map starts at world view */ });
  }, []);

  return { state, dispatch };
}
```

The hook reads last-known position from IndexedDB `settings` store (key: `lastPosition`) on mount. It does NOT create its own GPS watcher (position comes via props from page.tsx).
