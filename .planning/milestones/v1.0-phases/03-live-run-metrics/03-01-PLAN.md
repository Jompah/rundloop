function formatElapsed(ms: number): string {
  const totalSeconds = Math.floor(ms / 1000);
  const hours = Math.floor(totalSeconds / 3600);
  const minutes = Math.floor((totalSeconds % 3600) / 60);
  const seconds = totalSeconds % 60;
  if (hours > 0) {
    return `${hours}:${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;
  }
  return `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;
}

  <behavior>
    computeRollingPace:
    - Input: trace with 5 points over 30s covering ~150m -> returns ~200 sec/km
    - Input: trace with 1 point -> returns null
    - Input: trace with 2 points but 0m distance (same coords) -> returns null
    - Input: trace spanning 60s with windowMs=30000 -> only uses last 30s of points
    - Input: empty array -> returns null

    computeAveragePace:
    - Input: 5000m in 1500000ms (25min) -> returns 300 sec/km (5:00/km)
    - Input: 0m distance -> returns null
    - Input: less than 1000ms elapsed -> returns null

    formatPace:
    - Input: 330 sec/km, 'km' -> "5:30"
    - Input: 330 sec/km, 'miles' -> "8:51" (330 * 1.60934 = 531.08 -> 8:51)
    - Input: null, 'km' -> "--:--"

    formatMetricDistance:
    - Input: 3200m, 'km' -> "3.2"
    - Input: 3200m, 'miles' -> "2.0" (3200/1609.34 = 1.988 -> "2.0")
    - Input: 0m, 'km' -> "0.0"

    computeRemainingDistance:
    - Input: routeDistance=5000, coveredDistance=3200 -> 1800
    - Input: routeDistance=5000, coveredDistance=5500 -> 0 (clamped)
    - Input: routeDistance=5000, coveredDistance=0 -> 5000

    formatElapsed:
    - Input: 90000ms (1:30) -> "01:30"
    - Input: 3661000ms (1:01:01) -> "1:01:01"
    - Input: 0ms -> "00:00"
  </behavior>
  <implementation>
    Create src/lib/metrics.ts with these exports:

    1. `computeRollingPace(trace: FilteredPosition[], windowMs = 30_000): number | null`
       - Walk backwards from trace end to find first point within window
       - Sum haversine distances within window (import haversineMeters from storage.ts)
       - Guard: return null if < 2 points in window or < 1m distance
       - Return (elapsedSec / distMeters) * 1000 = seconds per km

    2. `computeAveragePace(distanceMeters: number, elapsedMs: number): number | null`
       - Guard: return null if distanceMeters < 10 or elapsedMs < 1000
       - Return (elapsedMs / 1000 / distanceMeters) * 1000

    3. `formatPace(paceSecsPerKm: number | null, units: 'km' | 'miles'): string`
       - Return '--:--' for null
       - Convert to per-mile if units === 'miles' (multiply by 1.60934)

    4. `formatMetricDistance(meters: number, units: 'km' | 'miles'): string`
       - Divide by 1000 for km, divide by 1609.34 for miles
       - Return with toFixed(1)

    5. `computeRemainingDistance(routeDistanceMeters: number, coveredDistanceMeters: number): number`
       - Return Math.max(0, routeDistanceMeters - coveredDistanceMeters)
